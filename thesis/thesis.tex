\catcode`_=\active
\gdef\code@activate@us{%
  \let_\textunderscore
  \catcode`\_=\active
}
\endgroup

  \fontsize{\f@size\code@factor}{\f@size\code@factor}\sffamily
  \scantokens{#1\endinput}%
  \endgroup
}
